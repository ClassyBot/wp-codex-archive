    Get terms for all custom taxonomies

    \n

    Place this function in your theme’s functions.php.

    \n
    \n
    \n
    \r\n/**\r\n * Get taxonomies terms links.\r\n *\r\n * @see get_object_taxonomies()\r\n */\r\nfunction wpdocs_custom_taxonomies_terms_links() {\r\n\t// Get post by post ID.\r\n\tif ( ! $post = get_post() ) {\r\n\t\treturn '';\r\n\t}\r\n\r\n\t// Get post type by post.\r\n\t$post_type = $post->post_type;\r\n\r\n\t// Get post type taxonomies.\r\n\t$taxonomies = get_object_taxonomies( $post_type, 'objects' );\r\n\r\n\t$out = array();\r\n\r\n\tforeach ( $taxonomies as $taxonomy_slug => $taxonomy ){\r\n\r\n\t\t// Get the terms related to post.\r\n\t\t$terms = get_the_terms( $post->ID, $taxonomy_slug );\r\n\r\n\t\tif ( ! empty( $terms ) ) {\r\n\t\t\t$out[] = "<h2>" . $taxonomy->label . "</h2>\\n<ul>";\r\n\t\t\tforeach ( $terms as $term ) {\r\n\t\t\t\t$out[] = sprintf( '<li><a href="%1$s">%2$s</a></li>',\r\n\t\t\t\t\tesc_url( get_term_link( $term->slug, $taxonomy_slug ) ),\r\n\t\t\t\t\tesc_html( $term->name )\r\n\t\t\t\t);\r\n\t\t\t}\r\n\t\t\t$out[] = "\\n</ul>\\n";\r\n\t\t}\r\n\t}\r\n\treturn implode( '', $out );\r\n}\r\n?>\r\n

    Another example of how to process the results of this call.

    \n

    This example retrieves the categories and tags for a post and uses wp_list_pluck() to efficiently creates a list of their names, without duplicates, and turns it into a string of keys suitable for an NITF feed.

    \n
    \n
    \n
    \r\n\t\t// Get a list of categories and extract their names\r\n\t\t$post_categories = get_the_terms( $post->ID, 'category' );\r\n\t\tif ( ! empty( $post_categories ) && ! is_wp_error( $post_categories ) ) {\r\n\t\t\t$categories = wp_list_pluck( $post_categories, 'name' );\r\n\t\t}\r\n\r\n\t\t// Get a list of tags and extract their names\r\n\t\t$post_tags = get_the_terms( $post->ID, 'post_tag' );\r\n\t\tif ( ! empty( $post_tags ) && ! is_wp_error( $post_tags ) ) {\r\n\t\t\t$tags = wp_list_pluck( $post_tags, 'name' );\r\n\t\t}\r\n\r\n\t\t// Combine Categories and tags, with category first\r\n\t\t$tagCats = array_merge( $categories, $tags );\r\n\r\n\t\t// Process to desired format\r\n\t\t$keyList = '<key-list>' .
