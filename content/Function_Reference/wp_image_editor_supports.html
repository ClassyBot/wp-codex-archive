Description[edit]

Tests whether there is an editor that supports a given mime type or methods.

Usage[edit]

$arg = array(
	'mime_type' => 'image/png',
	'methods' => array(
		'rotate',
		'resize',
		'save'
	)
);
$img_editor_test = wp_image_editor_supports($arg);
if ($img_editor_test !== false) {
	//mime_type and method(s) supported!
}
